// Smooth value animations
function animateValue(element, start, end, duration) {
    const range = end - start;
    const increment = range / (duration / 16);
    let current = start;
    
    const timer = setInterval(() => {
        current += increment;
        if (current >= end) {
            clearInterval(timer);
        }
        element.textContent = `$${current.toFixed(2)}`;
    }, 16);
}
